						 join "Zones" z on z."ZoneCategoryId" = zc."Id"

						 join "SeasonalTicketSeatings" sts  on s2."Id" = sts."SeatId" 
						 join "SeasonalTickets" st  on sts."SeasonalTicketId" = st."Id"
